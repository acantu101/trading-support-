#!/bin/bash
# Usage: ./monitor.sh /var/log/trading/app.log

LOG_FILE=$1
THRESHOLD=5
INTERVAL=10

if [ -z "$LOG_FILE" ]; then
    echo "Usage: $0 <log_file>"
    exit 1
fi

echo "Monitoring $LOG_FILE for CRITICAL events..."

while true; do
    # Get lines from last 60 seconds using tail (approx)
    COUNT=$(tail -n 1000 "$LOG_FILE" | grep "CRITICAL" | wc -l)

    if [ "$COUNT" -gt "$THRESHOLD" ]; then
        TIMESTAMP=$(date '+%Y-%m-%d %H:%M:%S')
        echo "[$TIMESTAMP] ⚠ ALERT: $COUNT CRITICAL events detected in $LOG_FILE"
    fi

    sleep $INTERVAL
done

P-03 Python: parse and analyze a log file Medium ⏱ 8 min ▾
Scenario: Write a Python script that reads a log file, counts errors by type, and prints a summary report. This is a classic support engineering task.
  • ✓ Read the log file line by line
  • ✓ Count occurrences of each log level (INFO, ERROR, WARN)
  • ✓ Print a sorted summary (most errors first)
  • ✓ Handle file not found gracefully
Think: collections.Counter, with open() as f, try/except FileNotFoundError, sorted(dict.items())
#!/usr/bin/env python3
# Run: python3 analyze_log.py /tmp/trading.log

import sys
from collections import Counter

def analyze_log(filepath):
    counts = Counter()
    errors = []

    try:
        with open(filepath) as f:
            for line in f:
                parts = line.split()
                if len(parts) >= 3:
                    level = parts[2]  # INFO / ERROR / WARN
                    counts[level] += 1
                    if level == "ERROR":
                        errors.append(line.strip())
    except FileNotFoundError:
        print(f"Error: file not found: {filepath}")
        sys.exit(1)

    print("\n=== Log Summary ===")
    for level, count in sorted(counts.items(),
                                  key=lambda x: x[1],
                                  reverse=True):
        print(f"{level:8s}: {count}")

    if errors:
        print(f"\nLast 3 errors:")
        for e in errors[-3:]:
            print(f"  {e}")

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: python3 analyze_log.py <logfile>")
        sys.exit(1)
    analyze_log(sys.argv[1])

import subprocess, time
from datetime import datetime

def is_running(process_name):
    result = subprocess.run(
        ['pgrep', '-x', process_name],
        capture_output=True
    )
    return result.returncode == 0

def monitor(process_name, interval=30):
    print(f"Monitoring '{process_name}' every {interval}s. Ctrl+C to stop.")
    try:
        while True:
            ts = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
            if is_running(process_name):
                print(f"[{ts}] ✓ {process_name} is running")
            else:
                print(f"[{ts}] ⚠ ALERT: {process_name} is NOT running!")

import os, sys
from datetime import datetime

def find_large_files(directory, min_mb=100):
    results = []
    for root, dirs, files in os.walk(directory):
        for fname in files:
            fpath = os.path.join(root, fname)

                mtime = os.path.getmtime(fpath)

        dt = datetime.fromtimestamp(mtime).strftime("%Y-%m-%d %H:%M")

T-02 Explain FIX protocol Medium ⏱ 4 min ▾
Scenario: Interviewer asks: "What is FIX protocol and what would you do if orders were being rejected with a FIX session disconnect?"
  • ✓ Define FIX protocol in plain language
  • ✓ Name 3 key FIX message types
  • ✓ Describe your troubleshooting steps for a FIX disconnect

WHAT IS FIX:
Financial Information eXchange — an industry-standard messaging protocol
for transmitting trade orders, executions, and market data between
trading firms and exchanges/brokers. It's text-based (tag=value pairs)
sent over TCP.

KEY MESSAGE TYPES:
- New Order Single (D) — place a new order
- Execution Report (8) — order fill/rejection/cancel confirmation
- Heartbeat (0) — keep-alive sent if no messages for X seconds
- Logon (A) / Logout (5) — session establish/teardown

FIX SESSION DISCONNECT TROUBLESHOOTING:
1. Check FIX engine logs — look for "Logout" message and reason code
   Common reasons: HeartbeatInterval timeout, CompID mismatch, bad sequence number

2. Check sequence numbers — FIX sessions track SeqNum
   If client/server are out of sync → ResendRequest or session reset needed

3. Check network — is the TCP connection alive?
   ss -tn dst 

4. Check credentials — SenderCompID/TargetCompID configured correctly?
   cat fix_config.cfg | grep CompID

5. Is the exchange FIX gateway up?
   ping exchange, check their status page

6. Session reset procedure:
   Some engines need a manual sequence number reset + reconnect.
   Coordinate with exchange's technical team.
T-03 What does a market maker do? Easy ⏱ 3 min ▾
Scenario: "Tell me what DRW does and why low latency matters to their business model."
  • ✓ Explain market making in simple terms
  • ✓ Explain the bid-ask spread
  • ✓ Connect latency to profitability

MARKET MAKING:
DRW simultaneously quotes prices to buy (bid) and sell (ask) an asset.
They profit from the spread — the difference between bid and ask.

Example: Stock at $100. DRW quotes bid $99.98 / ask $100.02.
They buy at 99.98 from sellers, sell at 100.02 to buyers. $0.04 profit.
Done millions of times per day across many assets.

WHY LATENCY MATTERS:
- DRW must constantly update quotes as the market moves.
- If they're slow to update, they get "picked off" — someone trades
  against a stale quote at an unfavorable price.

SECURITY GROUPS:

K-01 Explain Kafka architecture to an interviewer Easy ⏱ 4 min ▾
Scenario: "Walk me through how Kafka works. We use it heavily for internal messaging between our trading systems."
  • ✓ Define: broker, topic, partition, offset, producer, consumer

KAFKA ARCHITECTURE — interview-ready explanation:

BROKERS: Kafka servers that store and serve messages.
A cluster has multiple brokers for redundancy.

TOPICS: Named streams of records (like a table in a DB).
e.g., "trade-executions", "market-data", "risk-updates"

PARTITIONS: Topics are split into partitions for parallelism.
Messages within a partition are strictly ordered.
More partitions = more throughput (more consumers can work in parallel).

PRODUCERS: Write messages to a topic. Can specify a key —
messages with the same key always go to the same partition
(e.g., all AAPL trades → partition 0, ensuring order).

CONSUMERS: Read messages from topics. Track their position via OFFSET —
an integer saying "I've read up to message #1042."
Kafka retains messages for a configurable time (e.g., 7 days),
so consumers can replay from any offset.

CONSUMER GROUPS: Multiple consumers sharing work.
Each partition is assigned to exactly ONE consumer in a group.
Scale by adding consumers (up to # of partitions).

REPLICATION:
Each partition has 1 leader + N replicas on other brokers.
If the leader broker dies, a replica becomes leader.
For trading: replication.factor=3 means 3 copies.
acks=all means producer waits for ALL replicas to confirm →
zero data loss guarantee.

WHY THIS MATTERS FOR TRADING:
→ Trade events must not be lost (acks=all, replication)
→ Events must be processed in order (partitioning by symbol)
→ Failed consumers can replay from last committed offset
→ Multiple risk systems can consume the same trade feed independently

K-06 Kafka at-least-once vs exactly-once Medium ⏱ 4 min ▾
Scenario: "In a trading context, what's the difference between at-least-once and exactly-once delivery? Why does it matter if a trade event is processed twice?"
  • ✓ Define at-most-once, at-least-once, exactly-once
  • ✓ Explain the risk of duplicate trade processing
  • ✓ Explain how Kafka achieves exactly-once with idempotent producers

DELIVERY SEMANTICS:

AT-MOST-ONCE: Message sent once, no retry. If it fails, it's lost.
→ Fine for metrics/logs. NEVER for trade events.

AT-LEAST-ONCE: Message retried until acknowledged. May be processed 2x.
→ Risk of duplicate fills. A 100-share order processed twice = 200 shares.

EXACTLY-ONCE: Message delivered and processed precisely once.
→ Required for trade execution, P&L calculation, risk.

HOW KAFKA ACHIEVES EXACTLY-ONCE:

1. IDEMPOTENT PRODUCER (enable.idempotence=true):
   Each message gets a sequence number. Broker deduplicates
   retries with the same sequence number.

2. TRANSACTIONS (for consume-transform-produce pipelines):
   producer.initTransactions()
   producer.beginTransaction()
   → process message
   → produce result
   producer.sendOffsetsToTransaction(offsets, consumerGroup)
   producer.commitTransaction()
   
   If anything fails → abortTransaction() → entire batch rolled back.

3. CONSUMER SIDE: read_committed isolation level ensures consumers
   only see committed transactional messages.

INTERVIEW ANSWER:
"For trade execution events, we'd use idempotent producers (enable.idempotence=true)
and transactional semantics for the consume-process-produce pipeline.
This ensures a trade event updates our position exactly once — a duplicate
would corrupt our P&L and risk calculations."

OTHER CAUSES OF SUDDEN QUERY SLOWNESS:

1. STALE STATISTICS
   Optimizer uses row count estimates to pick a plan.
   After a large data load, estimates are wrong → bad plan chosen.
   Fix (PostgreSQL): ANALYZE trades;
   Fix (MySQL): ANALYZE TABLE trades;

2. INDEX WAS DROPPED OR BECAME BLOATED
   Someone dropped an index during a migration.
   Or after millions of updates, the index is fragmented.
   Fix: REINDEX TABLE trades; (PG) or OPTIMIZE TABLE trades; (MySQL)

3. LOCK CONTENTION
   A long-running write transaction holds a lock.
   Other queries queue behind it.
   Fix: pg_stat_activity / SHOW PROCESSLIST (MySQL) to find blocker,
        kill the blocking query if safe.

4. TABLE GREW PAST MEMORY THRESHOLD
   Query used to fit in shared_buffers, now spills to disk.
   Fix: increase work_mem (PG), or partition the table by date.

5. MISSING QUERY CACHE / CACHE INVALIDATED
   After a large write, query cache flushed → cold cache.
   Fix: usually self-resolving, or pre-warm with a dummy query.

#!/usr/bin/env python3
# Airflow REST API health checker
# Requires: Airflow 2.x with REST API enabled

import urllib.request, json
from datetime import datetime, timedelta, timezone

AIRFLOW_URL = "http://localhost:8080/api/v1"
AUTH = ("admin", "admin")  # base64 encoded in real usage

CRITICAL_DAGS = [
    "market_data_ingestion",
    "forecasting_pipeline",
    "risk_calculation",
    "trade_reconciliation",
]

def get_dag_runs(dag_id, hours=24):
    since = (datetime.now(timezone.utc) - timedelta(hours=hours)).isoformat()
    url = f"{AIRFLOW_URL}/dags/{dag_id}/dagRuns?state=failed&start_date_gte={since}"
    import base64
    token = base64.b64encode(f"{AUTH[0]}:{AUTH[1]}".encode()).decode()
    req = urllib.request.Request(url,
        headers={"Authorization": f"Basic {token}",
                 "Content-Type": "application/json"})
    try:
        with urllib.request.urlopen(req, timeout=10) as r:
            return json.loads(r.read())["dag_runs"]
    except Exception as e:
        print(f"  ⚠ API error for {dag_id}: {e}")
        return []

def check_pipelines():
    ts = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    print(f"\n[{ts}] Checking {len(CRITICAL_DAGS)} critical DAGs...\n")
    alerts = 0
    for dag_id in CRITICAL_DAGS:
        runs = get_dag_runs(dag_id)
        if runs:
            alerts += 1
            for run in runs:
                start = run.get('start_date', 'unknown')[:19]
                print(f"  🔴 FAILED: {dag_id}")
                print(f"     Run ID : {run['dag_run_id']}")
                print(f"     Started: {start}")
                print(f"     State  : {run['state']}\n")
        else:
            print(f"  ✅ OK: {dag_id}")
    if alerts == 0:
        print("\nAll pipelines healthy.")
    else:
        print(ff"\n⚠ {alerts} DAG(s) require attention.")

check_pipelines()
